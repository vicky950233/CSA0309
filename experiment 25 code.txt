#include <stdio.h>

int parent[10];

int find(int i) {
    while(parent[i]) i=parent[i];
    return i;
}

int uni(int i,int j) {
    if(i!=j){ parent[j]=i; return 1; }
    return 0;
}

int main() {
    int n,cost[10][10],i,j,u,v,a,b,min,ne=1,total=0;
    printf("Enter number of vertices: ");
    scanf("%d",&n);

    printf("Enter cost matrix (9999 for no edge):\n");
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++) {
            scanf("%d",&cost[i][j]);
            if(cost[i][j]==0) cost[i][j]=9999;
        }

    while(ne < n) {
        min=9999;
        for(i=1;i<=n;i++)
            for(j=1;j<=n;j++)
                if(cost[i][j] < min) {
                    min=cost[i][j]; a=u=i; b=v=j;
                }
        u=find(u); v=find(v);
        if(uni(u,v)) {
            printf("Edge %d: %d -> %d cost=%d\n",ne++,a,b,min);
            total+=min;
        }
        cost[a][b]=cost[b][a]=9999;
    }
    printf("Minimum cost = %d\n",total);
    return 0;
}
