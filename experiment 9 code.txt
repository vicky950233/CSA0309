#include <stdio.h>
int main() {
    int a[50], n, i, key, low, high, mid, found=0;
    printf("Enter size: ");
    scanf("%d",&n);
    printf("Enter elements (sorted): ");
    for(i=0;i<n;i++) scanf("%d",&a[i]);
    printf("Enter number to search: ");
    scanf("%d",&key);
    low=0; high=n-1;
    while(low<=high){
        mid=(low+high)/2;
        if(a[mid]==key){ printf("Found at position %d",mid+1); found=1; break; }
        else if(key<a[mid]) high=mid-1;
        else low=mid+1;
    }
    if(!found) printf("Not Found");
    return 0;
}
