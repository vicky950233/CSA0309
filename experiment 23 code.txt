#include <stdio.h>
#define INF 9999
#define MAX 10

int main() {
    int n, cost[MAX][MAX], dist[MAX], visited[MAX]={0};
    int i,j,start,u,v,min;
    
    printf("Enter number of vertices: ");
    scanf("%d",&n);

    printf("Enter cost matrix (9999 for no edge):\n");
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            scanf("%d",&cost[i][j]);

    printf("Enter starting vertex: ");
    scanf("%d",&start);

    for(i=0;i<n;i++) dist[i]=cost[start][i];
    dist[start]=0; visited[start]=1;

    for(i=1;i<n;i++) {
        min=INF;
        for(j=0;j<n;j++)
            if(!visited[j] && dist[j]<min) { min=dist[j]; u=j; }
        visited[u]=1;
        for(v=0;v<n;v++)
            if(!visited[v] && dist[u]+cost[u][v]<dist[v])
                dist[v]=dist[u]+cost[u][v];
    }

    printf("Shortest distances from %d:\n",start);
    for(i=0;i<n;i++)
        printf("%d -> %d = %d\n",start,i,dist[i]);
}
